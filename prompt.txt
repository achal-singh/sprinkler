Also, I intend to execute the authorisation transaction only when I see an EOA is not delegated to my BatchTransfer.sol contract so maybe we can use the following code as a reference:



const BATCH_TRANSFER_ADDRESS = '0xac2605...' // your deployed BatchTransfer address

function isDelegatedToBatchTransfer(code) {

  // EIP-7702 delegation designator: 0xef0100 + 20-byte address

  const prefix = '0xef0100'

  

  if (!code.startsWith(prefix)) return false

  

  const delegateAddress = '0x' + code.slice(prefix.length)

  return delegateAddress.toLowerCase() === BATCH_TRANSFER_ADDRESS.toLowerCase()

}

// Usage

const code = await provider.getCode(eoaAddress)

if (code === '0x') {

  console.log('Regular EOA, no delegation')

} else if (isDelegatedToBatchTransfer(code)) {

  console.log('Delegated to our BatchTransfer contract')

} else if (code.startsWith('0xef0100')) {

  const otherContract = '0x' + code.slice(8)

  console.log(`Delegated to a different contract: ${otherContract}`)

} else {

  console.log('This is a regular contract, not an EOA delegation')

}
